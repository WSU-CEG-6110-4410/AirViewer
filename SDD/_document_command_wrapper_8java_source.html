391  @Override
392  public AbstractDocumentCommand execute() {
393  AbstractDocumentCommand result = null;
394  PDAnnotation candidate = null;
395 
396  assert 5 == arguments.size();
397 
398  int pageNumber = parseInt(arguments.get(0));
399  float x = parseFloat(arguments.get(1));
400  float y = parseFloat(arguments.get(2));
401  float dx = parseFloat(arguments.get(3));
402  float dy = parseFloat(arguments.get(4));
403 
404  if (null == annotations || 0 == annotations.size()) {
405  try {
406  // We have to find the annotation to move
407  PDPage page = owner.wrappedDocument.getPage(pageNumber);
408  List<PDAnnotation> oldAnnotations = page.getAnnotations();
409  candidate = owner.getLastAnnotationAtPoint(oldAnnotations, x, y);
410  } catch (IOException ex) {
411 // Logger.getLogger(DocumentCommandWrapper.class.getName()).log(Level.SEVERE, null, ex);
412  }
413  } else {
414  // We can just delete annotation
415  candidate = annotations.get(0);
416  }
417 
418  if (null != candidate) {
419 
420  ArrayList<String> newArgs = new ArrayList<>(arguments);
421  newArgs.set(3, Float.toString(-dx));
422  newArgs.set(4, Float.toString(-dy));
423  ArrayList<PDAnnotation> candidateList = new ArrayList<>();
424  candidateList.add(candidate);
425  result = new MoveAnnotationDocumentCommand(owner, candidateList, newArgs);
426 
427  PDRectangle position = candidate.getRectangle();
428  position.setLowerLeftX(position.getLowerLeftX() + dx);
429  position.setLowerLeftY(position.getLowerLeftY() + dy);
430  position.setUpperRightX(position.getUpperRightX() + dx);
431  position.setUpperRightY(position.getUpperRightY() + dy);
432  candidate.setRectangle(position);
433  }
434 
435  return result;
436  }

621 
622  int pageNumber = parseInt(arguments.get(0));
623  List<PDAnnotation> oldAnnotations;
624  List<PDAnnotation> selectedAnnotations = owner.getSelectedAnnotations();
625  AbstractDocumentCommand result = null;
626 
627  if (0 < selectedAnnotations.size()) {
628  try {
629  PDPage page = owner.wrappedDocument.getPage(pageNumber);
630  oldAnnotations = page.getAnnotations();
631  result = new ReplaceAnnotationDocumentCommand(owner, new ArrayList<>(oldAnnotations), arguments);
632  for (PDAnnotation a : selectedAnnotations) {
633  List<PDAnnotation> itemsToRemove = oldAnnotations.stream().filter(
634  p -> p.getAnnotationName().equals(a.getAnnotationName())).collect(Collectors.toCollection(ArrayList::new));
635 
636  if (null != itemsToRemove) {
637  for (PDAnnotation pa : itemsToRemove) {
638  oldAnnotations.remove(pa);
639  }
640  }
641  }
642  page.setAnnotations(oldAnnotations);
643  owner.deselectAll();
644  } catch (IOException ex) {
645 // Logger.getLogger(DocumentCommandWrapper.class.getName()).log(Level.SEVERE, null, ex);
